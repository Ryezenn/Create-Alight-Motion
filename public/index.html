Petunjuk Integrasi Bot:

  1. Salin API Key pribadi Anda yang ada di atas. Jaga kerahasiaannya.
  2. Gunakan template Node.js atau Python di bawah sebagai dasar pengembangan bot Anda.
  3. Ganti teks placeholder YOUR_API_KEY di bagian headers request dengan API Key Anda.
  4. Ganti teks placeholder MASUKKAN_URL_WEBSITE_DI_SINI dengan domain website generator ini (misal: ${window.location.origin}).
  5. Setiap request aktivasi sukses melalui API akan memotong 1 Kredit dari akun Anda (Admin dibebaskan/tidak dipotong kredit).
// 1. Kirim Link Verifikasi
const axios = require('axios');
const API_URL = window.location.origin;

axios.post(`${API_URL}/api/v1/bot/send-link`, {
    email: 'target@email.com'
}, {
    headers: { 'X-API-Key': 'YOUR_API_KEY' }
}).then(res => console.log(res.data))
  .catch(err => console.error(err.response?.data || err.message));

// 2. Terapkan Lisensi Premium
axios.post(`${API_URL}/api/v1/bot/activate`, {
    email: 'target@email.com',
    magicLink: 'https://alightcreative.com/...'
}, {
    headers: { 'X-API-Key': 'YOUR_API_KEY' }
}).then(res => console.log(res.data))
  .catch(err => console.error(err.response?.data || err.message));
# Template Python Bot Integration
import requests
API_URL = "MASUKKAN_URL_WEBSITE_DI_SINI" # Contoh: https://site.com

headers = { 'X-API-Key': 'YOUR_API_KEY' }

# 1. Kirim Link Verifikasi
r1 = requests.post(f"{API_URL}/api/v1/bot/send-link", json={
    "email": "target@email.com"
}, headers=headers)
print(r1.json())

# 2. Terapkan Lisensi Premium
r2 = requests.post(f"{API_URL}/api/v1/bot/activate", json={
    "email": "target@email.com",
    "magicLink": "https://alightcreative.com/..."
}, headers=headers)
print(r2.json())
